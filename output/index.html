// Box Component
// Source: shallot_components/src/box.rs

use maud::{html, Markup, Render};
use shallot_foundation::theme::Theme;

pub struct Box {
    pub text: String,
    pub variant: Option<String>,
}

impl Box {
    pub fn new(text: impl Into<String>) -> Self {
        Self {
            text: text.into(),
            variant: None,
        }
    }

    pub fn variant(mut self, variant: impl Into<String>) -> Self {
        self.variant = Some(variant.into());
        self
    }
}

impl Render for Box {
    fn render(&self) -> Markup {
        html! {
            div class=(format!("sh-{}", stringify!(box)) {
                (self.text)
            }
        }
    }
}
// 1. Add to your Cargo.toml:
// [dependencies]
// shallot_components = "0.1"

// 2. Import the component:
use shallot_components::box::Box;

// 3. Use in your Maud template:
html! {
    (Box::new("Example").render())
}
